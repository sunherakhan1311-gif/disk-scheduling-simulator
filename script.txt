function calculate() {

    let queue = document.getElementById("queue").value
        .split(",")
        .map(Number);

    let head = parseInt(document.getElementById("head").value);

    let algo = document.getElementById("algo").value;

    let output = "";

    if (algo === "fcfs") {

        let total = 0;
        let current = head;

        output += "FCFS Disk Scheduling\n\n";

        queue.forEach(req => {

            total += Math.abs(req - current);

            output += current + " -> " + req + "\n";

            current = req;
        });

        output += "\nTotal Seek Time = " + total;
    }

    else if (algo === "sstf") {

        let requests = [...queue];
        let current = head;
        let total = 0;

        output += "SSTF Disk Scheduling\n\n";

        while (requests.length > 0) {

            let closest = requests.reduce((prev, curr) =>
                Math.abs(curr - current) < Math.abs(prev - current)
                ? curr : prev
            );

            total += Math.abs(closest - current);

            output += current + " -> " + closest + "\n";

            current = closest;

            requests.splice(requests.indexOf(closest), 1);
        }

        output += "\nTotal Seek Time = " + total;
    }

    document.getElementById("output").innerText = output;
}
